from numpy import ceil def cups_of_coffee_desired(cups): cups_to_grams = 250 grams_of_water_needed = cups * cups_to_grams return grams_of_water_needed def chemex(coffee_grams,water_ratio): if coffee_grams > 70: raise Exception("Too much coffee") water = coffee_grams * water_ratio if water > 900: raise Exception("Too much water, risk of overflow") bloom = 70 pour = water - bloom if pour < 0: raise Exception("Not enough water") return [coffee_grams, bloom, pour, water] def brew(cups, water_ratio): grams_of_water = cups_of_coffee_desired(cups) grams_of_coffee = ceil(grams_of_water/water_ratio) coffee = chemex(grams_of_coffee, water_ratio) return coffee def sub(*args,**kwargs): cups_of_coffee = Element('cups').value water_ratio = Element('ratio').value coffee_beans, bloom, pour, water = brew(int(cups_of_coffee), int(water_ratio)) result_place = Element('output') result_place.write(f""" {cups_of_coffee} cups of coffee with a bean to water ratio of 1:{water_ratio} """) instruct = Element('instructions') instruct.write(f""" If you want {cups_of_coffee} cup(s) of coffee: 1. Pour {bloom}g of water and wait 60 seconds 6. Pour {pour}g of water in a circular motion without touching the filter NOTE: Some suggest pouring {pour}g in three phases of {pour/3:0.0f}g each 7.
